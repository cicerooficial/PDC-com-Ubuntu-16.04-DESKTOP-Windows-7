Modo global que vai definir as configurações do Samba

cola tudo abaixo

LINK: https://www.hardware.com.br/tutoriais/samba-pdc/

[global]
	workgroup = ch
	netbios name = ch1
	server string = ch SAMBA PDC

	domain master = yes
	domain logons = yes

	security = user
	encrypt passwords = yes
	enable privileges = yes

	preferred master = yes
	local master = yes
	os level = 100
	wins support = yes

	dns proxy = no
	log file = /var/log/samba/log.%m
	max log size = 1000
	syslog = 0
	panic action = /usr/share/samba/panic-action

	ntlm auth = yes
	lanman auth = yes
	client ntlmv2 auth = yes

[netlogon]
	comment = Servico de Logon
	path = /var/samba/netlogon
	read only = yes
	browseable = no

[homes]
	valid users = %S
	create mask = 0700
	directory mask = 0700
	browseable = no
